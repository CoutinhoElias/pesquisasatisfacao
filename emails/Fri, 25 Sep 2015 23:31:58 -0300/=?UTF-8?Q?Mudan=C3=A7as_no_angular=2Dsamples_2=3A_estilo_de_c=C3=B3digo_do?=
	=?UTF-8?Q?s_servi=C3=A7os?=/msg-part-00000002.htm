Outra mudança que alguns podem ter notado é no estilo de definir os serviços.

Não estou mais usando o estilo abaixo (estilo A)

angular.module('githubmodel').factory('GithubModel', function(GithubApi, $log){
var gm = {
username_input: '',
buscando_usuarios: false,
usuarios_encontrados: [],
mostra_usuarios_encontrados: false,
user: null,
buscando_repositorios: false,
repos_encontrados: [],
repo: null, //setado externamente via binding
buscando_issues: false,
};

gm.busca_usuarios_pelo_nome = function(){...}
gm.escolhe_usuario = function(user){...}
gm.busca_repos_do_usuario = function(){...}
gm.busca_mais_issues = function(){...}

return gm;
});



Ao invés etou usando o estilo a seguir (estilo B)


angular.module('githubmodel').factory('GithubModel', function(GithubApi, $log){
var gm = {
username_input: '',
buscando_usuarios: false,
usuarios_encontrados: [],
mostra_usuarios_encontrados: false,
user: null,
buscando_repositorios: false,
repos_encontrados: [],
repo: null, //setado externamente via binding
buscando_issues: false,
};

angular.extend(gm, {
busca_usuarios_pelo_nome: busca_usuarios_pelo_nome,
escolhe_usuario: escolhe_usuario,
busca_repos_do_usuario: busca_repos_do_usuario,
busca_mais_issues: busca_mais_issues,
reset_repo: reset_repo,
toggle_expand: toggle_expand,
carrega_arquivo: carrega_arquivo,
carrega_comentarios_da_issue: carrega_comentarios_da_issue,
});

return gm;

function busca_usuarios_pelo_nome(){...}
function escolhe_usuario(user){...}
function busca_repos_do_usuario(){...}
function busca_mais_issues(){...}
});


O efeito é igual:o objeto gm vai ter as mesmas coisas. 
Pra quem não sabe, angular.extend é uma função utilitária do angular

angular.extend(a, b); //copia todos os atributos do objeto a pro objeto b

Estou dando preferência ao estilo B porquê dá pra ver logo no começo os atributos e métodos do objeto. (no estilo A, eu tenho que scrolar pra ver os metodos depois.)
Eu até poderia colocar a lista de métodos junto com a lista de atributos (na declaração do var gm = ...). O angular.extend() é uma frescura a mais pra explicitar "estado aqui, comportamento ali"
