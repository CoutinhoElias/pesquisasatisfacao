Voce precisa fazer a interação entre os modelos, ForeignKey, ManyToMany, OneToOne o que melhor se encaixa no seu problema. E depois colocar a ordem que você deseja no forms.
Exemplo:


models.py

class Pessoa(models.Model):

    name = CharField()

    telefone = ForeignKey('Telefone') / ManyToManyField('Telefone') / OneToOneField('Telefone')


class Telefone(models.Model):

    telefone = CharField()


forms.py

class PessoaForms(forms.Form):

    model = Pessoa

    class Meta:

        field = ['nome', 'telefone']
