Quando você envia o email ele só vai aparecer no console por que seu EMAIL_BACKEND esta configurado para aparecer no console.

EMAIL_BACKEND = 'django.core.mail.backends.console.EmailBackend'

Caso queira que realmente envie, mude para smtp:

EMAIL_BACKEND = 'django.core.mail.backends.smtp.EmailBackend'

subscriptions/views.py
from django.contrib import messages
from django.core import mail
from django.http import HttpResponseRedirect, HttpResponse
from django.shortcuts import render
from django.template.loader import render_to_string
from eventex.subscriptions.forms import SubscriptionForm


def subscribe(request):
    if request.method == 'POST':
        form = SubscriptionForm(request.POST)

        if form.is_valid():
            body = render_to_string('subscriptions/subscription_email.txt',
                                    form.cleaned_data)

            mail.send_mail('Confirmação de inscrição',
                           body,
                           'rafaelcb21@gmail.com',
                           ['rafaelcb21@gmail.com', form.cleaned_data['email']])

            messages.success(request, 'Inscrição realizada com sucesso!')
            return HttpResponseRedirect('/inscricao/')
        else:
            return render(request, 'subscriptions/subscription_form.html',
                          {'form': form})
    else:
        context = {'form': SubscriptionForm()}
        return render(request, 'subscriptions/subscription_form.html', context)
