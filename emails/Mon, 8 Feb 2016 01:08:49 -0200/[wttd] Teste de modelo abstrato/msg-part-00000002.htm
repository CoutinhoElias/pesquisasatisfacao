É válido fazer teste de modelo abstrato?

class Address(models.Model):
    address = models.CharField(_(u'endereço'), max_length=80)
    # ...

    class Meta:
        abstract = True

class AddressModelTest(TestCase):

    def setUp(self):
        self.obj = Address(
            address=u'Avenida Paulista, 1605',
            complement='Apto 303',
            district=u'Bela Vista',
            city=u'São Paulo',
            uf='SP',
            cep='01311200')
        self.obj.save()

    def test_create(self):
        self.assertTrue(Address.objects.exists())

    def test_str(self):
        self.assertEqual('Avenida Paulista, 1605', str(self.obj))
